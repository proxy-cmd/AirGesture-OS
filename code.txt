#if defined(ESP8266)
  #include <ESP8266WiFi.h>
  #include <ESP8266WebServer.h>
  ESP8266WebServer server(80);
#elif defined(ESP32)
  #include <WiFi.h>
  #include <WebServer.h>
  WebServer server(80);
#else
  #error "Use ESP8266 or ESP32 board"
#endif

// ================= USER SETTINGS =================
const char* WIFI_SSID = "Himanshu2";
const char* WIFI_PASSWORD = "himanshu";

// Fixed dispense run time (ms)
const int DISPENSE_RUN_MS = 20000;  // 20 seconds

// Motor speed (0-255)
const int MOTOR_SPEED = 220;
// =================================================

#if defined(ESP8266)
  const int PIN_IN1 = D1;  // GPIO5
  const int PIN_IN2 = D2;  // GPIO4
  const int PIN_ENA = D5;  // GPIO14 (PWM)
#elif defined(ESP32)
  const int PIN_IN1 = 26;
  const int PIN_IN2 = 27;
  const int PIN_ENA = 25;
#endif

#if defined(ESP32)
  const int PWM_CHANNEL = 0;
  const int PWM_FREQ = 5000;
  const int PWM_RES = 8;
#endif

void setMotorPWM(int value) {
#if defined(ESP8266)
  analogWrite(PIN_ENA, value);     // 0..255
#elif defined(ESP32)
  ledcWrite(PWM_CHANNEL, value);   // 0..255
#endif
}

void stopMotor() {
  digitalWrite(PIN_IN1, LOW);
  digitalWrite(PIN_IN2, LOW);
  setMotorPWM(0);
}

void rotateForward(int durationMs) {
  digitalWrite(PIN_IN1, HIGH);
  digitalWrite(PIN_IN2, LOW);
  setMotorPWM(MOTOR_SPEED);
  delay(durationMs);
  stopMotor();
}

int parseAmountFromBody(const String& body) {
  int i = body.indexOf("\"amount\"");
  if (i < 0) return 0;

  int colon = body.indexOf(':', i);
  if (colon < 0) return 0;

  int p = colon + 1;
  while (p < body.length() && (body[p] == ' ' || body[p] == '\t')) p++;

  String n = "";
  while (p < body.length() && isDigit(body[p])) {
    n += body[p++];
  }

  return n.length() ? n.toInt() : 0;
}

void handleHealth() {
  server.send(200, "application/json", "{\"status\":\"ok\",\"device\":\"chutta-mate-controller\"}");
}

void handleDispense() {
  String body = server.arg("plain");
  int amount = parseAmountFromBody(body);
  if (amount <= 0) amount = 10;

  int runMs = DISPENSE_RUN_MS;

  Serial.println("[DISPENSE] Trigger received");
  Serial.print("Amount: "); Serial.println(amount);
  Serial.print("Run ms: "); Serial.println(runMs);

  rotateForward(runMs);

  server.send(200, "application/json", "{\"status\":\"success\",\"motor\":\"rotated\"}");
}

void connectWifi() {
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  Serial.print("Connecting WiFi");
  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 40) {
    delay(500);
    Serial.print(".");
    tries++;
  }
  Serial.println();

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("WiFi connected");
    Serial.print("ESP IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("WiFi failed. Check SSID/PASSWORD and 2.4GHz network.");
  }
}

void setup() {
  Serial.begin(115200);
  delay(200);

  pinMode(PIN_IN1, OUTPUT);
  pinMode(PIN_IN2, OUTPUT);
  pinMode(PIN_ENA, OUTPUT);

#if defined(ESP8266)
  analogWriteRange(255);
#elif defined(ESP32)
  ledcSetup(PWM_CHANNEL, PWM_FREQ, PWM_RES);
  ledcAttachPin(PIN_ENA, PWM_CHANNEL);
#endif

  stopMotor();
  connectWifi();

  server.on("/health", HTTP_GET, handleHealth);
  server.on("/dispense", HTTP_POST, handleDispense);
  server.begin();

  Serial.println("HTTP server started");
  Serial.println("GET  /health");
  Serial.println("POST /dispense");
}

void loop() {
  server.handleClient();
}
