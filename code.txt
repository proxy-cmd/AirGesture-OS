/*
  Proxy Bank -> ESP32 Dispenser Bridge

  Flow:
  1) Flask backend sends POST to ESP32 /dispense after payment success.
  2) ESP32 rotates motor via motor driver.

  Hardware (example L298N + BO motor):
  ESP32 GPIO26 -> IN1
  ESP32 GPIO27 -> IN2
  ESP32 GPIO25 -> ENA (PWM)
  GND shared between ESP32 and motor driver

  IMPORTANT:
  - Motor power should come from external supply (not ESP32 5V pin).
  - Keep grounds common (ESP32 GND <-> driver GND).
*/

#include <WiFi.h>
#include <WebServer.h>

// ======== YOUR WIFI ========
const char* WIFI_SSID = "YOUR_WIFI_NAME";
const char* WIFI_PASSWORD = "YOUR_WIFI_PASSWORD";

// Optional simple token check (set same value in backend request header if you add one).
const char* API_KEY = "proxybank-demo";

// ======== MOTOR PINS (edit if needed) ========
const int MOTOR_IN1 = 26;
const int MOTOR_IN2 = 27;
const int MOTOR_ENA = 25; // PWM-capable pin

// PWM config
const int PWM_CHANNEL = 0;
const int PWM_FREQ = 5000;
const int PWM_RESOLUTION = 8; // 0..255
const int MOTOR_SPEED = 210;  // tune this

WebServer server(80);

void stopMotor() {
  digitalWrite(MOTOR_IN1, LOW);
  digitalWrite(MOTOR_IN2, LOW);
  ledcWrite(PWM_CHANNEL, 0);
}

void rotateForward(int durationMs) {
  digitalWrite(MOTOR_IN1, HIGH);
  digitalWrite(MOTOR_IN2, LOW);
  ledcWrite(PWM_CHANNEL, MOTOR_SPEED);
  delay(durationMs);
  stopMotor();
}

int extractAmountFromBody(const String& body) {
  // very lightweight extraction for JSON like: {"amount":20,...}
  int idx = body.indexOf("\"amount\"");
  if (idx < 0) return 0;
  int colon = body.indexOf(':', idx);
  if (colon < 0) return 0;

  int i = colon + 1;
  while (i < body.length() && (body[i] == ' ' || body[i] == '\t')) i++;

  String num = "";
  while (i < body.length() && isDigit(body[i])) {
    num += body[i];
    i++;
  }

  if (num.length() == 0) return 0;
  return num.toInt();
}

void handleHealth() {
  server.send(200, "application/json", "{\"status\":\"ok\",\"device\":\"esp32-dispenser\"}");
}

void handleDispense() {
  // Optional API key validation
  String key = server.header("X-API-Key");
  if (key.length() > 0 && key != API_KEY) {
    server.send(401, "application/json", "{\"status\":\"failed\",\"message\":\"Invalid API key\"}");
    return;
  }

  String body = server.arg("plain");
  int amount = extractAmountFromBody(body);
  if (amount <= 0) amount = 10;

  // Duration mapping (tune for your tread structure)
  int runMs = 350 + (amount * 15);
  if (runMs > 2200) runMs = 2200;

  Serial.println("[DISPENSE] Request received");
  Serial.print("Amount: ");
  Serial.println(amount);
  Serial.print("Motor run (ms): ");
  Serial.println(runMs);

  rotateForward(runMs);

  server.send(200, "application/json", "{\"status\":\"success\",\"motor\":\"rotated\"}");
}

void connectWifi() {
  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  Serial.print("Connecting WiFi");
  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 40) {
    delay(500);
    Serial.print(".");
    tries++;
  }
  Serial.println();

  if (WiFi.status() == WL_CONNECTED) {
    Serial.print("WiFi connected. ESP32 IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("WiFi connection failed. Rebooting in 5 sec...");
    delay(5000);
    ESP.restart();
  }
}

void setup() {
  Serial.begin(115200);

  pinMode(MOTOR_IN1, OUTPUT);
  pinMode(MOTOR_IN2, OUTPUT);
  ledcSetup(PWM_CHANNEL, PWM_FREQ, PWM_RESOLUTION);
  ledcAttachPin(MOTOR_ENA, PWM_CHANNEL);
  stopMotor();

  connectWifi();

  server.on("/health", HTTP_GET, handleHealth);
  server.on("/dispense", HTTP_POST, handleDispense);
  server.begin();

  Serial.println("HTTP server started");
  Serial.println("GET  /health");
  Serial.println("POST /dispense");
}

void loop() {
  server.handleClient();
}
